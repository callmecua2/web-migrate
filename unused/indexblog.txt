<script>
    (function() {
        // Cast NodeListOf<Element> menjadi NodeListOf<HTMLElement>
        const tabs = document.querySelectorAll('.tab-btn') as NodeListOf<HTMLElement>;
        const cards = document.querySelectorAll('.article-card') as NodeListOf<HTMLElement>;
        const grid = document.getElementById('articleGrid') as HTMLElement | null;
        const emptyState = document.getElementById('emptyState') as HTMLElement | null;

        // Validasi elemen dasar untuk mencegah error script crash dan memperjelas null-check ke TS
        if (!grid || !emptyState) return;

        function filterArticles(category: string) {
            let visibleCount = 0;

            // Pastikan grid tidak null (sudah divalidasi di atas, tapi TS butuh kepastian jika scope berbeda)
            grid!.classList.add('filtering');

            cards.forEach(card => {
                const cardCat = card.dataset.category;
                const shouldShow = (category === 'all' || cardCat === category);

                if (shouldShow) {
                    card.style.display = 'flex';
                    visibleCount++;
                    
                    // PERBAIKAN: Paksa reflow agar animasi re-trigger bekerja valid di semua browser
                    card.style.animation = 'none';
                    card.offsetWidth; // Membaca properti ini memaksa browser merender ulang perubahan
                    card.style.animation = '';
                } else {
                    card.style.display = 'none';
                }
            });

            // PERBAIKAN: Tampilkan empty state murni berdasarkan jumlah kartu yang terlihat
            if (visibleCount === 0) {
                emptyState!.style.display = 'block';
            } else {
                emptyState!.style.display = 'none';
            }

            // Hapus class filtering setelah transisi selesai
            setTimeout(() => {
                grid!.classList.remove('filtering');
            }, 250);
        }

        tabs.forEach(tab => {
            tab.addEventListener('click', function(e) {
                // Hindari penggunaan 'this' implisit dengan mendefinisikan target secara eksplisit
                const currentTab = e.currentTarget as HTMLElement;

                // Jangan jalankan ulang fungsi jika tab yang diklik sudah aktif (efisiensi)
                if (currentTab.classList.contains('active')) return;

                tabs.forEach(t => t.classList.remove('active'));
                currentTab.classList.add('active');

                const category = currentTab.dataset.category || 'all';
                filterArticles(category);
            });
        });

        // Inisialisasi awal
        filterArticles('all');
    })();
</script>
